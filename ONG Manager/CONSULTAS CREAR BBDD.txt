// CREAR LA TABLA USUARIOS

CREATE TABLE USERS (
    ID       INTEGER     PRIMARY KEY AUTOINCREMENT,
    USUARIO  STRING (10),
    PASSWORD STRING (10),
    PERFIL   INTEGER (1) 
);


// CREAR LA TABLA CURSOS

CREATE TABLE CURSOS (
    ID            INTEGER     PRIMARY KEY AUTOINCREMENT,
    TIPO          STRING (10),
    NOMBRE        STRING,
    FECHAINICIO   DATE,
    FECHAFINAL    DATE,
    HORAS         INTEGER (6),
    HORASDIA      INTEGER (2),
    PERIODICIDAD  STRING (20),
    SESIONES      INTEGER (6),
    COMPLETADO    STRING (2),
    PROFESOR      INTEGER,
    OBSERVACIONES STRING
);


// CREAR LA TABLA ALUMNOS

CREATE TABLE ALUMNOS (
    ID         INTEGER     PRIMARY KEY AUTOINCREMENT,
    NOMBRE     STRING,
    APELLIDO1  STRING,
    APELLIDO2  STRING,
    NIF        STRING (20),
    PAIS       STRING (20),
    EDAD       INTEGER (3),
    SEXO       STRING (10),
    TELEFONO1  INTEGER (9),
    TELEFONO2  INTEGER (9),
    EMAIL      STRING,
    DIRECCION1 STRING,
    DIRECCION2 STRING,
    POBLACION  STRING,
    PROVINCIA  STRING,
    OBSERVACIONES STRING
);


// CREAR LA TABLA SOCIOS

CREATE TABLE SOCIOS (
    ID            INTEGER     PRIMARY KEY AUTOINCREMENT,
    NOMBRE        STRING,
    APELLIDO1     STRING,
    APELLIDO2     STRING,
    NIF           STRING (20),
    PAIS          STRING (20),
    EDAD          INTEGER (3),
    SEXO          STRING (10),
    TELEFONO1     INTEGER (9),
    TELEFONO2     INTEGER (9),
    EMAIL         STRING,
    DIRECCION1    STRING,
    DIRECCION2    STRING,
    POBLACION     STRING,
    PROVINCIA     STRING,
    FECHADEALTA   DATE,
    FECHADEBAJA   DATE,
    SITUACION     STRING,
    TIPODECUOTA   STRING,
    PERIODICIDAD  STRING,
    IMPORTE       DOUBLE,
    CUENTA        STRING,
    OBSERVACIONES STRING
);


// CREAR LA TABLA ALUMNOSCURSO ((RELACIONA A LOS ALUMNOS CON LOS CURSOS EN LOS QUE SE HA INSCRITO))

CREATE TABLE ALUMNOSCURSO(
    ID          INTEGER PRIMARY KEY AUTOINCREMENT,
    IDCURSO    INTEGER,
    IDALUMNO    INTEGER,
    COMPLETADO STRING (2)
);


// CREAR LA TABLA ARCHIVOSALUMNOS (( GUARDA LA RUTA DONDE SE ALMACENAN LOS ARCHIVOS ))

CREATE TABLE ARCHIVOSALUMNOS (
    ID          INTEGER PRIMARY KEY AUTOINCREMENT,
    IDALUMNO    INTEGER,
    RUTAARCHIVO STRING
);


// CREAR LA TABLA ARCHIVOSSOCIOS (( GUARDA LA RUTA DONDE SE ALMACENAN LOS ARCHIVOS ))

CREATE TABLE ARCHIVOSALUMNOS (
    ID          INTEGER PRIMARY KEY AUTOINCREMENT,
    IDSOCIO    INTEGER,
    RUTAARCHIVO STRING
);
